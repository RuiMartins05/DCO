lis<Migrante> listMigr;

private class Migrante {
        string nome;
        string contacto;
        int numPessoas;
        String[] fam;
        int counter = 0;

        public Migrante(nome, contacto) {
            this.nome = nome;
            this.contacto = contacto;
        }

        private registaFamilia(num) {
            this.numPessoas = num;
            this.fam = new String[num];
            this.fam[counter] = this.nome;
            counter++;
        }

        private indicaCabecaCasal(nome, contacto) {
            this.nome = nome;
            this.contacto = contacto;
        }

        private indicaMembro(nome) {
            if (this.fam.size() < this.numPessoas) {
                this.fam[counter] = nome;
                counter++;
            }
        }
}
