.2Caso de Uso: Procurar Ajuda
Ator Principal: Migrante

Pré-Condições: N/A

Pós-Condições: Registadas ajudas no sistema e sms enviado aos voluntarios das ajudas.

--

CONTRATOS: 

    op: registaMigrante(nome, contacto)
        pré-condições:
            - N/A.
        pós-condições:
            - seja m:Migrante tal que m.nome = nome e
                 m.contacto = contacto. 

    op: registaFamilia(num)
        pré-condições:
            - num > 1;
        pós-condições:
            - seja m:Migrante corrente;
            - m.numPessoasFam = num;
            - é criado fam:String[num]. 

    op: indicaCabecaCasal(nome, contacto)
        pré-condições:
            - existe m corrente.
        pós-condições:
            - m.nome = nome e m.contacto = contacto.
            
    op: indicaMembro(nome)
        pré-condições:
            - existe m corrente.
        pós-condições:
            - adicionado nome a fam.

    op: getListaRegioes()
        pré-condições:
            - existe m corrente;
            - existe lr:ListaRegioes.
        pós-condições:
            - N/A.

    op: indicaRegiao(regiao)
        pré-condições:
            - existe m corrente;
            - existe lr:ListaRegioes;
            - la:ListaAjudas associada a regiao.
        pós-condições:
            - N/A.

    op: indicaAjuda(la)
        pré-condições:
            - existe m corrente;
            - existe la:ListaAjudas;
            - criada (ajudasSelecionadas:list<Ajuda>) se ajudasSelecionadas == null. 
        pós-condições:
            - seja a:Ajuda tal que a:Ajuda pertence a la:ListaAjudas,
                 m.ajudaCorrente = a 

    op: registaAjuda()
        pré-condições:
            - existe m corrente;
            - existe ajudaCorrente;
            - existe ajudasSelecionadas:list<Ajuda> corrente.
        pós-condições:
            - ajudaCorrente:Ajuda registada em ajudasSelecionadas:list<Ajuda>.

    op: confirmarAjudas()
        pré-condições:
            - existe m corrente;
            - existe ajudasSelecionadas:list<Ajuda> corrente.
        pós-condições:
            - pretendeConfirmar = true.

    op: registaConfirmacao()
        pré-condições:
            - existe m corrente;
            - existe ajudasSelecionadas:list<Ajuda> corrente.
        pós-condições:
            - toda a listaAjuda[i] que exista em ajudasSelecionadas
                tem listaAjuda[i].migranteAtual = m 

- Extensão 5a:

    op: pretendeNotificacao(regiao)
        pré-condições:
            - existe m corrente;
            - existe lr:ListaRegioes empty.
            - existe pretendeNotificacao:list<Migrante>
        pós-condições:
            - migrante associado a regiao
            - adicionado migrante a pretendeNotificacao.
